        session.navigationDelegate = object : GeckoSession.NavigationDelegate {
            override fun onNewSession(s: GeckoSession, uri: String): GeckoResult<GeckoSession>? {
                Log.d(TAG, "onNewSession called: $uri")
                val newSession = GeckoSession()
                setupSession(newSession, serverPort)
                addTab(newSession, uri)
                
                val result = GeckoResult<GeckoSession>()
                result.then { createdSession ->
                    Log.d(TAG, "New session completed, switching GeckoView")
                    Handler(Looper.getMainLooper()).post {
                        geckoView.setSession(createdSession)
                    }
                    createdSession
                }
                result.complete(newSession)
                return result
            }
        }
